--- SACAR TODAS LAS CATEGORÍAS ASOCIADAS A UN CAMPEONATO (fem > masc > mixto) ---

SELECT campeonato.NOMBRE, categoria.SEXONIVEL FROM campeonato INNER JOIN (campeonato_categoria JOIN categoria ON campeonato_categoria.ID_CATEGORIA = categoria.ID_CATEGORIA) ON campeonato.ID_CAMPEONATO = campeonato_categoria.ID_CAMPEONATO ORDER BY categoria.SEXONIVEL
